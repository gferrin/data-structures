template <typename T>
int BST<T>::height(BSTNode *t) const
{
	if( t == NULL ){
		return -1;
	}
	int left = height( t->left );
	int right = height( t->right );

	if( left > right ){
		return left + 1;
	} else {
		return right + 1;
	}
}

It does not check to see if a node is Null before calling the function again,
and it does this twice for every node being considered. Therefore this 
function has a time compleity of O( N^2 )
